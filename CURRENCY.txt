  <script>
    function convertCurrency() {
      const amount = document.getElementById('amount').value;
      const fromCurrency = document.querySelector('input[name="fromCurrency"]:checked').value;
      const toCurrency = 'INR'; // Fixed to INR

      if (!amount) {
        alert("Please enter an amount to convert.");
        return;
      }

      // Hardcoded exchange rates (values as examples; update as needed)
      const exchangeRates = {
        USD: 82.75, // 1 USD to INR
        JPY: 0.61,  // 1 JPY to INR
        EUR: 89.43, // 1 EUR to INR
        GBP: 101.85, // 1 GBP to INR
        AUD: 55.65, // 1 AUD to INR
        CAD: 60.23  // 1 CAD to INR
      };

      // Get the exchange rate for the selected currency
      const rate = exchangeRates[fromCurrency];
      if (!rate) {
        alert("Exchange rate not available for the selected currency.");
        return;
      }

      // Perform the conversion
      const convertedAmount = (amount * rate).toFixed(2);

      // Display the result
      document.getElementById('result').innerHTML = `${amount} ${fromCurrency} = ${convertedAmount} INR`;
    }
  </script>
